TPL - is a html template syster in javascript

CDN:

https://cdn.rawgit.com/frentsel/_tpl/master/tpl.min.js


Example: 
      

    User name: {user.firstName}
,
    age: {user.age}
,
    mother: {user.parent.mother.firstName}
Result:
Javascript code: 
      
var data = {
    user:  {
        firstName: "Alex",
        age: 27,
        parent: {
            mother: {
                firstName: "Samantha"
            }
        }
    }
};

var el = document.getElementById('text');
var html = el.innerHTML;
el.innerHTML = tpl(html, data);

// or jQuery variant
var $text = $('#text');
var html = tpl($text.html(), data);
$text.html(html);
